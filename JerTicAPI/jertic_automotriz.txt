USE jertic_automotriz;

-- =========================
-- CATÁLOGOS
-- =========================

CREATE TABLE estado_general (
    id_estado INT IDENTITY PRIMARY KEY,
    nombre NVARCHAR(50) UNIQUE
);

CREATE TABLE tipo_usuario (
    id_tipo_usuario INT IDENTITY PRIMARY KEY,
    nombre NVARCHAR(50) UNIQUE
);

CREATE TABLE tipo_producto (
    id_tipo_producto INT IDENTITY PRIMARY KEY,
    nombre NVARCHAR(100) UNIQUE
);

CREATE TABLE tipo_servicio (
    id_tipo_servicio INT IDENTITY PRIMARY KEY,
    nombre NVARCHAR(100) UNIQUE,
    descripcion NVARCHAR(MAX)
);

CREATE TABLE especialidad_tecnico (
    id_especialidad INT IDENTITY PRIMARY KEY,
    nombre NVARCHAR(100) UNIQUE
);

-- =========================
-- USUARIOS
-- =========================

CREATE TABLE usuario (
    id_usuario INT IDENTITY PRIMARY KEY,
    nombre NVARCHAR(100),
    apellido NVARCHAR(100),
    correo NVARCHAR(150) UNIQUE,
    telefono NVARCHAR(20),
    token NVARCHAR(255),
    fecha_registro DATETIME DEFAULT GETDATE(),
    id_tipo_usuario INT,
    id_estado INT,
    FOREIGN KEY (id_tipo_usuario) REFERENCES tipo_usuario(id_tipo_usuario),
    FOREIGN KEY (id_estado) REFERENCES estado_general(id_estado)
);

CREATE TABLE usuario_cliente (
    id_cliente INT IDENTITY PRIMARY KEY,
    id_usuario INT UNIQUE,
    FOREIGN KEY (id_usuario) REFERENCES usuario(id_usuario)
);

CREATE TABLE usuario_tecnico (
    id_tecnico INT IDENTITY PRIMARY KEY,
    id_usuario INT UNIQUE,
    id_especialidad INT,
    FOREIGN KEY (id_usuario) REFERENCES usuario(id_usuario),
    FOREIGN KEY (id_especialidad) REFERENCES especialidad_tecnico(id_especialidad)
);

-- =========================
-- PRODUCTOS
-- =========================

CREATE TABLE productos (
    id_producto INT IDENTITY PRIMARY KEY,
    nombre NVARCHAR(150),
    descripcion NVARCHAR(MAX),
    precio DECIMAL(10,2),
    stock INT,
    fecha_creacion DATETIME DEFAULT GETDATE(),
    id_tipo_producto INT,
    FOREIGN KEY (id_tipo_producto) REFERENCES tipo_producto(id_tipo_producto)
);

-- =========================
-- CITAS
-- =========================

CREATE TABLE cita (
    id_cita INT IDENTITY PRIMARY KEY,
    id_cliente INT,
    fecha DATE,
    hora TIME,
    motivo NVARCHAR(MAX),
    id_estado INT,
    FOREIGN KEY (id_cliente) REFERENCES usuario_cliente(id_cliente),
    FOREIGN KEY (id_estado) REFERENCES estado_general(id_estado)
);

-- =========================
-- SERVICIOS
-- =========================

CREATE TABLE servicio (
    id_servicio INT IDENTITY PRIMARY KEY,
    id_cita INT,
    id_tecnico INT,
    descripcion NVARCHAR(MAX),
    costo DECIMAL(10,2),
    fecha_inicio DATETIME,
    fecha_fin DATETIME,
    id_estado INT,
    id_tipo_servicio INT,
    FOREIGN KEY (id_cita) REFERENCES cita(id_cita),
    FOREIGN KEY (id_tecnico) REFERENCES usuario_tecnico(id_tecnico),
    FOREIGN KEY (id_estado) REFERENCES estado_general(id_estado),
    FOREIGN KEY (id_tipo_servicio) REFERENCES tipo_servicio(id_tipo_servicio)
);

CREATE TABLE historial_servicios (
    id_historial INT IDENTITY PRIMARY KEY,
    id_servicio INT,
    descripcion NVARCHAR(MAX),
    fecha_evento DATETIME,
    FOREIGN KEY (id_servicio) REFERENCES servicio(id_servicio)
);

-- =========================
-- VENTAS
-- =========================

CREATE TABLE venta (
    id_venta INT IDENTITY PRIMARY KEY,
    id_cliente INT,
    fecha DATETIME,
    total DECIMAL(10,2),
    FOREIGN KEY (id_cliente) REFERENCES usuario_cliente(id_cliente)
);

CREATE TABLE detalle_venta (
    id_detalle INT IDENTITY PRIMARY KEY,
    id_venta INT,
    id_producto INT,
    cantidad INT,
    precio_unitario DECIMAL(10,2),
    subtotal DECIMAL(10,2),
    FOREIGN KEY (id_venta) REFERENCES venta(id_venta),
    FOREIGN KEY (id_producto) REFERENCES productos(id_producto)
);

-- =========================
-- TRIGGER EVITAR STOCK NEGATIVO
-- =========================

GO
CREATE TRIGGER trg_validar_y_restar_stock
ON detalle_venta
INSTEAD OF INSERT
AS
BEGIN
    SET NOCOUNT ON;

    -- Validar stock suficiente
    IF EXISTS (
        SELECT 1
        FROM inserted i
        JOIN productos p ON p.id_producto = i.id_producto
        WHERE p.stock < i.cantidad
    )
    BEGIN
        RAISERROR ('Stock insuficiente para realizar la venta', 16, 1);
        RETURN;
    END

    -- Insertar detalle
    INSERT INTO detalle_venta (id_venta, id_producto, cantidad, precio_unitario, subtotal)
    SELECT id_venta, id_producto, cantidad, precio_unitario, subtotal
    FROM inserted;

    -- Descontar stock
    UPDATE p
    SET p.stock = p.stock - i.cantidad
    FROM productos p
    JOIN inserted i ON p.id_producto = i.id_producto;

END
GO
